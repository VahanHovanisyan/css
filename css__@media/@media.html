@media

Հրահանգ, որը թույլ է տալիս տարբեր ոճեր սահմանել էկրանի տարբեր պարամետրերի համար՝ լայնություն, բարձրություն, կողմնորոշում և նույնիսկ սարքի տեսակը 

	@media սարքի տեսակը (սարքի բնութագիրը) {
		․selector{
			հատկություն։ առժեք;
		}
	}

@media (max-width:767px) { .selector{ background-color: red; color: white; } }

@media (max-height:767px) { .selector{ background-color: yellow; } }

@media (min-width:768px) { .selector{ background-color: blue; color:white; } }

@media (min-height:768px) { .selector{ background-color: green; } }

@media (min-width:576px) and (max-width:991px) { .selector{ background-color: orange; } }

@media (pointer: fine) and (hover: hover) { .selector:hover { background-color: pink; } }

@media (pointer: coarse) { .selector { background-color: greenyellow; } }

@media (orientation: landscape) { .selector{ background-color: purple; color: white; } }

@media (orientation: portrait) { .selector{ background-color: aqua; } }

@media print { .selector{ color: red; } }
